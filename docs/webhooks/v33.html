from dataclasses import dataclass

@dataclass
class WebhookV33:
    webhook_id: str
    secret_key: str
    action: str
    license_id: int
    additional_data: dict
    payload: dict

    def payload_data_class(self):
        ''' Returns payload's data class for webhook's action. '''
        return action_to_data_class_mapping_v_33[self.action]
